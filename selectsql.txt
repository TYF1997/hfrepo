select l.*, p.*, lcont.*, lduser.*
  from Piplineinfo p
  left join Lccont l
    on p.transno = l.transno
  left join Lcnotconclusion lcont
    on lcont.contno = l.proposalcontno
  left join Lduser lduser
    on lduser.emplid = p.rmid
 where p.autodropflag = '0'
   and (p.branch in (select pb.branchcode
                       from Piplinebranch pb
                      where pb.staffid = 'ybt') or p.branch is null)
 order by lduser.emplid ASC, p.piplineno desc
